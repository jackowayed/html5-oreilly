phonegap                                  // The root directory of the PhoneGap project
phonegap/android                          // The directory for Android app files
phonegap/blackberry                       // The directory for BlackBerry app files
phonegap/iphone                           // The directory for iPhone app files
phonegap/iphone/beep.wav                  // Sound file played when app needs to beep
phonegap/iphone/Default.png               // Image file used as the startup graphic at app launch
phonegap/iphone/icon.png                  // Image file used as home screen icon
phonegap/iphone/Info.plist                // Settings file; edit from within Kilo.xcodeproj
phonegap/iphone/Kilo.xcodeproj            // The project file; open this file in Xcode to get started
phonegap/iphone/MainWindow.xib            // Interface Builder file used by PhoneGap; there's no need to touch this
phonegap/iphone/PhoneGap.plist            // Settings file; edit from within Kilo.xcodeproj
phonegap/iphone/Settings.plist            // Settings file; edit from within Kilo.xcodeproj
phonegap/iphone/www                       // The directory that contains the web app
phonegap/iphone/www/index.html            // Main page of the application
phonegap/iphone/www/jqtouch/jqtouch.css   // Low level jQTouch style settings referenced in index.html
phonegap/iphone/www/jqtouch/jqtouch.js    // jQTouch library referenced in index.html
phonegap/iphone/www/jqtouch/jquery.js     // jQuery library referenced in index.html
phonegap/iphone/www/kilo.css              // App specific CSS referenced in index.html
phonegap/iphone/www/kilo.js               // App specific JavaScript referenced in index.html
phonegap/iphone/www/themes/jqt/img        // Directory of images referenced in theme.css
phonegap/iphone/www/themes/jqt/theme.css  // Theme-specific CSS referenced in index.html
phonegap/javascripts                      // Directory of JavaScript files used by PhoneGap
phonegap/winmo                            // The directory for Windows Mobile app files
